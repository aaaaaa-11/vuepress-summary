// 方法3
function unique (arr) {
  let newArr = []
  for (let i = 0; i < arr.length; i++) {
    let isUnique = true
    // 从当前索引向后遍历，查找是否元素和当前元素相同
    for (let j = i + 1; j < arr.length; j++) {
      if (arr[i] === arr[j]) {
        isUnique = false
        break
      }
    }
    // 如果后面没有元素和当前元素相同，则当前元素唯一，push进新数组
    isUnique && newArr.push(arr[i])
  }
  return newArr
}
